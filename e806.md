## SmallShawn95
* Cpp 第一種解法：
```cpp
#include <iostream>
#include <map>
using namespace std;

int main()
{
    pair<int, int> num[1001];
    for(int i = 0; i < 1001; i++){
        num[i] = {i, 0};
    }
    int n, x, y;
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> x >> y;
        num[x].second += y;
    }
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> x >> y;
        num[x].second += y;
    }
    bool no = true;
    for(int i = 1000; i >= 0; i--){
        if(num[i].second != 0){
            no = false;
            cout << num[i].first << ':' << num[i].second << '\n';
        }
    }
    if(no){
        cout << "NULL!\n";
    }
    return 0;
}
```

* Cpp 第二種解法：
```cpp
#include <iostream>
using namespace std;

int main()
{
    int n, x, y;
    int num[1001] = {0};
    cin >> n;
    while(n--){
        cin >> x >> y;
        num[x] = y;
    }
    cin >> n;
    while(n--){
        cin >> x >> y;
        num[x] += y;
    }
    bool no = true;
    for(int i = 1000; i >= 0; i--){
        if(num[i] != 0){
            no = false;
            cout << i << ':' << num[i] << '\n';
        }
    }
    if(no){
        cout << "NULL!\n";
    }
    return 0;
}
```

## Not Exist
```cpp
#include<bits/stdc++.h>
using namespace std;

#define endl '\n'

int main(){
    ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
    //創一個裝多項式各項的map
    map<int,int> polynomial;
    int n,x,y;
    cin>>n;
    //把第一個式子，依序分類放到map
    while(n--){
        cin>>x>>y;
        polynomial[x]+=y;
    }
    cin>>n;
    //把第二個式子，依序分類放到map
    while(n--){
        cin>>x>>y;
        polynomial[x]+=y;
    }
    //設一個bool值用來判斷多項式是否為0項
    bool b=0;
    //設一個interator當反向迭代器讓他從最大項數開始數到最前面的項數
    for(auto inter=polynomial.rbegin();inter!=polynomial.rend();inter++){
        if((*inter).second){
            cout<<(*inter).first<<":"<<(*inter).second<<endl;
            b=1;
        }
    }
    if(!b)
        cout<<"NULL!\n";
}
```
